import { Badge } from '@kairo-ui/react';

## การใช้งาน

```tsx
import { Badge } from '@kairo-ui/react';

export function Example() {
  return <Badge intent="success">Active</Badge>;
}
```

## Props

| Prop        | Type                                                              | Default     | Description                             |
| ----------- | -------------------------------------------------------------------- | ------------ | ------------------------------------------ |
| `variant`   | `'solid' \| 'soft' \| 'outline'`                                    | `'solid'`    | รูปแบบการแสดงผลของ badge                   |
| `intent`    | `'default' \| 'primary' \| 'success' \| 'warning' \| 'danger'`       | `'default'`  | สีเชิงความหมาย (intent) ของ badge          |
| `className` | `string`                                                             | —            | รวมกับคลาสพื้นฐาน `kairo-badge`            |
| ...         | `HTMLAttributes<HTMLSpanElement>`                                    | —            | props ดั้งเดิมอื่น ๆ ของ `<span>` ทั้งหมดจะถูกส่งผ่านไป |

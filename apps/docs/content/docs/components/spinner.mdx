import { Spinner } from '@kairo-ui/react';

## Usage

```tsx
import { Spinner } from '@kairo-ui/react';

export function Example() {
  return <Spinner aria-label="Loading" />;
}
```

## Props

| Prop         | Type                               | Default     | Description                                          |
| ------------ | ----------------------------------- | ------------ | ------------------------------------------------------ |
| `size`       | `'sm' \| 'md' \| 'lg'`              | `'md'`       | Size of the spinner.                                   |
| `aria-label` | `string`                             | `'Loading'`  | Accessible name announced to assistive technology.      |
| `className`  | `string`                             | —            | Merged with the base `kairo-spinner` class.             |
| ...          | `HTMLAttributes<HTMLSpanElement>`    | —            | All other native `<span>` props are passed through.     |

import { Spinner } from '@kairo-ui/react';

## การใช้งาน

```tsx
import { Spinner } from '@kairo-ui/react';

export function Example() {
  return <Spinner aria-label="Loading" />;
}
```

## Props

| Prop         | Type                               | Default     | Description                                          |
| ------------ | ----------------------------------- | ------------ | ------------------------------------------------------ |
| `size`       | `'sm' \| 'md' \| 'lg'`              | `'md'`       | ขนาดของ spinner                                         |
| `aria-label` | `string`                             | `'Loading'`  | ชื่อที่เข้าถึงได้ (accessible name) ที่ประกาศให้ assistive technology ทราบ |
| `className`  | `string`                             | —            | รวมกับคลาสพื้นฐาน `kairo-spinner`                        |
| ...          | `HTMLAttributes<HTMLSpanElement>`    | —            | รับ props ดั้งเดิมของ `<span>` ที่เหลือทั้งหมด            |

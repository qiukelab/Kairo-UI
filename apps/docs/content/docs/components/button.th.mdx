import { Button } from '@kairo-ui/react';

## การใช้งาน

```tsx
import { Button } from '@kairo-ui/react';

export function Example() {
  return <Button>Click me</Button>;
}
```

## Props

| Prop        | Type                             | Default   | Description                                  |
| ----------- | --------------------------------- | --------- | --------------------------------------------- |
| `variant`   | `'solid' \| 'outline' \| 'ghost'` | `'solid'` | รูปแบบการแสดงผลของปุ่ม                          |
| `size`      | `'sm' \| 'md' \| 'lg'`            | `'md'`    | ขนาดของปุ่ม                                     |
| `type`      | `'button' \| 'submit' \| 'reset'` | `'button'`| attribute `type` ของ `<button>` ดั้งเดิม        |
| `className` | `string`                          | —         | รวมกับคลาสพื้นฐาน `kairo-btn`                   |
| ...         | `ButtonHTMLAttributes<HTMLButtonElement>` | —  | รับ props ดั้งเดิมของ `<button>` ที่เหลือทั้งหมด (เช่น `onClick`, `disabled`, `aria-*`) |

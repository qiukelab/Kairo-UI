import { Input } from '@kairo-ui/react';

## การใช้งาน

```tsx
import { Input } from '@kairo-ui/react';

export function Example() {
  return <Input placeholder="Email address" aria-label="Email address" />;
}
```

## Props

| Prop           | Type                                     | Default   | Description                                                       |
| -------------- | ------------------------------------------ | ---------- | --------------------------------------------------------------------- |
| `type`         | `string`                                   | `'text'`   | attribute `type` ของ `<input>` ดั้งเดิม (เช่น `'email'`, `'password'`) |
| `aria-invalid` | `boolean \| 'true' \| 'false'`             | —          | ทำเครื่องหมายว่า input ไม่ถูกต้อง จัดสไตล์ด้วยเส้นขอบสีแจ้งเตือน (danger) |
| `disabled`     | `boolean`                                   | —          | ป้องกันการโต้ตอบของผู้ใช้ จัดสไตล์ด้วยความทึบที่ลดลง                   |
| `className`    | `string`                                    | —          | รวมกับคลาสพื้นฐาน `kairo-input`                                       |
| ...            | `InputHTMLAttributes<HTMLInputElement>`     | —          | รับ props ดั้งเดิมของ `<input>` ที่เหลือทั้งหมด (เช่น `value`, `onChange`, `placeholder`) |
